\documentclass{standalone}
\usepackage{bubbleplot}

\definecolor{2017}{rgb}{1, 0.5, 0.5}
\definecolor{2018}{rgb}{0.5, 1, 0.5}
\definecolor{2019}{rgb}{0.5, 0.5, 1}
\pgfplotsset{
    colormap={yearcolor}{
      color=(2017),
      color=(2018),
      color=(2019),
    }
}

\begin{document}

\bubbleplot[%
    width=7mm,
    height=1cm,
    xmin=-10,
    xmax=8,
    enlarge y limits=0.3,
    ylabel=Snakes,
    year x shift=-1.4cm,
    year y shift=3.5cm,
    x left label=Fishes,
    x left label shift=3cm,
    x right label=Birds,
    x right label shift=2cm,
]{Fishes_Snakes_Birds.csv}{2017, 2018, 2019}
\end{document}
